<script>
  // ====== State ======
  const PLAY_BARS = 8;
  const bars = Array.from({length: PLAY_BARS}, () => ({A:"", B:""})); // A=前半, B=後半
  let selectedBar = 1; // 1..8
  let selectedHalf = "A"; // A or B
  let compose = ""; // build chord string by taps

  // ====== UI Build ======
  function renderBars(){
    const grid = document.getElementById("barGrid");
    grid.innerHTML = "";
    for(let i=1;i<=PLAY_BARS;i++){
      const row = document.createElement("div");
      row.className = "bar";

      const idx = document.createElement("div");
      idx.className = "idx";
      idx.textContent = i;

      const slotA = document.createElement("div");
      slotA.className = "slot" + (selectedBar===i && selectedHalf==="A" ? " active" : "");
      slotA.onclick = ()=>{ selectedBar=i; selectedHalf="A"; renderAll(); };
      slotA.innerHTML = `<div class="txt">${escapeHtml(bars[i-1].A || "（前半）")}</div><div class="badge">A</div>`;

      const slotB = document.createElement("div");
      slotB.className = "slot" + (selectedBar===i && selectedHalf==="B" ? " active" : "");
      slotB.onclick = ()=>{ selectedBar=i; selectedHalf="B"; renderAll(); };
      slotB.innerHTML = `<div class="txt">${escapeHtml(bars[i-1].B || "（後半）")}</div><div class="badge">B</div>`;

      row.appendChild(idx);
      row.appendChild(slotA);
      row.appendChild(slotB);
      grid.appendChild(row);
    }
  }

  function renderPads(){
    const roots = ["C","C#","D","D#","E","F","F#","G","G#","A","A#","B","D♭","E♭","G♭","A♭","B♭"];
    const rootPad = document.getElementById("rootPad");
    rootPad.innerHTML = "";
    roots.forEach(r=>{
      const b=document.createElement("button");
      b.className="btn";
      b.textContent=r;
      b.onclick=()=>{ compose += r; renderCompose(); };
      rootPad.appendChild(b);
    });

    const quals = [
      "", "m", "7", "m7", "maj7", "M7", "dim", "aug",
      "sus4", "m7-5", "add9", "9", "6", "m6"
    ];
    const qualPad = document.getElementById("qualPad");
    qualPad.innerHTML = "";
    quals.forEach(q=>{
      const b=document.createElement("button");
      b.className="btn";
      b.textContent = q==="" ? "（なし）" : q;
      b.onclick=()=>{ compose += q; renderCompose(); };
      qualPad.appendChild(b);
    });
  }

  function renderHalfButtons(){
    document.getElementById("btnHalfA").classList.toggle("primary", selectedHalf==="A");
    document.getElementById("btnHalfB").classList.toggle("primary", selectedHalf==="B");
  }

  function renderCompose(){
    // show compose in the currently selected slot badge area by temporarily replacing badge
    // (軽量：ページタイトルを使う)
    document.title = compose ? `入力中: ${compose}` : "Progression Entry (Tap UI)";
  }

  function renderAll(){
    renderBars();
    renderHalfButtons();
    renderCompose();
  }

  function setHalf(h){
    selectedHalf = h;
    renderAll();
  }

  function clearSelected(){
    bars[selectedBar-1][selectedHalf] = "";
    compose = "";
    renderAll();
  }

  function backspace(){
    compose = compose.slice(0, Math.max(0, compose.length-1));
    renderCompose();
  }

  function addSplit(){
    compose += "│";
    renderCompose();
  }

  function addSlash(){
    compose += "/";
    renderCompose();
  }

  function applyChord(){
    const v = compose.trim();
    if(!v) return;
    bars[selectedBar-1][selectedHalf] = v;
    compose = "";
    renderAll();
  }

  function resetAll(){
    for(let i=0;i<PLAY_BARS;i++){
      bars[i].A=""; bars[i].B="";
    }
    compose="";
    document.getElementById("outJson").textContent="{}";
    renderAll();
  }

  function buildChordForBar(i){
    const a = (bars[i].A||"").trim();
    const b = (bars[i].B||"").trim();
    if(a && b) return `${a}│${b}`;
    if(a) return a;
    if(b) return b; // 変則入力も許容
    return "";
  }

  function generateJson(){
    const chords = [];
    for(let i=0;i<PLAY_BARS;i++){
      chords.push(buildChordForBar(i));
    }

    const obj = {
      artist: val("artist"),
      title: val("title"),
      part: val("part"),
      ERA: val("era"),
      rank: numOrText("rank"),
      year: numOrText("year"),
      original_key: val("okey"),
      quiz_key: val("qkey") || "C",
      tag1: val("tag1"),
      tag2: val("tag2"),
      tag3: val("tag3"),
      tag4: val("tag4"),
      comment: val("comment"),
      chords: chords,
      // degrees は後で入力する前提で空。必要になったらここもタップ化できます。
      degrees: ["","","","","","","",""]
    };

    document.getElementById("outJson").textContent = JSON.stringify(obj, null, 2);
  }

  async function copyJson(){
    const txt = document.getElementById("outJson").textContent;
    try{
      await navigator.clipboard.writeText(txt);
      toast("コピーしました");
    }catch(e){
      // fallback
      const ta=document.createElement("textarea");
      ta.value=txt;
      document.body.appendChild(ta);
      ta.select();
      document.execCommand("copy");
      document.body.removeChild(ta);
      toast("コピーしました（互換モード）");
    }
  }

  function toast(msg){
    const d=document.createElement("div");
    d.textContent=msg;
    d.style.position="fixed";
    d.style.left="50%";
    d.style.bottom="24px";
    d.style.transform="translateX(-50%)";
    d.style.background="#111";
    d.style.color="#fff";
    d.style.padding="10px 14px";
    d.style.borderRadius="999px";
    d.style.fontSize="14px";
    d.style.opacity="0.95";
    d.style.zIndex="9999";
    document.body.appendChild(d);
    setTimeout(()=>{ d.remove(); }, 1200);
  }

  function escapeHtml(s){
    return String(s??"")
      .replaceAll("&","&amp;")
      .replaceAll("<","&lt;")
      .replaceAll(">","&gt;")
      .replaceAll('"',"&quot;")
      .replaceAll("'","&#39;");
  }
  function val(id){
    const el=document.getElementById(id);
    return (el && el.value!=null) ? String(el.value).trim() : "";
  }
  function numOrText(id){
    const s = val(id);
    if(s==="" ) return "";
    const n = Number(s);
    return Number.isFinite(n) ? n : s;
  }

  // init
  renderPads();
  renderAll();
</script>
